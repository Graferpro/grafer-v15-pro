// public/app.js içindeki fetchData fonksiyonunu bununla değiştir: async function fetchData() { try { // Artık /api/get-keys yerine direkt proxy'ye istek atıyoruz // Bu adres bizim güvenli sunucumuz const res = await fetch('/api/proxy'); if(!res.ok) throw new Error(`Sunucu Hatası: ${res.status}`); const data = await res.json(); if(data.error) throw new Error(data.error); state.rates = data.results; // Veriyi kaydet console.log("✅ Veriler proxy üzerinden başarıyla çekildi."); } catch(e) { console.error("Veri Çekme Hatası:", e); // Hata olursa yine de demo verilerle çalışmaya devam et } }
